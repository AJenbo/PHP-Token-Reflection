102:      *

105:      * @throws \TokenReflection\Exception\Parse If the constant name could not be determined

107:     protected function parseName(Stream $tokenStream)
108:     {
109:         try {
110:             if ($tokenStream->is(T_CONST)) {
111:                 $tokenStream->skipWhitespaces();
112:             }
113: 
114:             if (!$tokenStream->is(T_STRING)) {
115:                 throw new Exception\Parse('The constant name could not be determined.', Exception\Parse::PARSE_ELEMENT_ERROR);
116:             }
117: 
118:             if (null === $this->namespaceName || $this->namespaceName === ReflectionNamespace::NO_NAMESPACE_NAME) {
119:                 $this->name = $tokenStream->getTokenValue();
120:             } else {
121:                 $this->name = $this->namespaceName . '\\' . $tokenStream->getTokenValue();
122:             }

124:             $tokenStream->skipWhitespaces();

126:             return $this;

155:     /**
156:      * Parses the constant value.
157:      *
158:      * @param \TokenReflection\Stream $tokenStream Token substream
159:      * @param \TokenReflection\IReflection $parent Parent reflection object
160:      * @return \TokenReflection\ReflectionConstant
161:      * @throws \TokenReflection\Exception\Parse If the constant value could not be determined
162:      */
163:     private function parseValue(Stream $tokenStream, IReflection $parent)
164:     {
165:         try {
166:             if (!$tokenStream->is('=')) {
167:                 throw new Exception\Parse('Could not find the definition start.', Exception\Parse::PARSE_ELEMENT_ERROR);
168:             }
169: 
170:             $tokenStream->skipWhitespaces();
171: 
172:             static $acceptedStrings = array('true' => true, 'false' => true, 'null' => true);
173:             static $acceptedTokens = array('-' => true, '+' => true, T_STRING => true, T_NS_SEPARATOR => true, T_CONSTANT_ENCAPSED_STRING => true, T_DNUMBER => true, T_LNUMBER => true, T_DOUBLE_COLON => true);
174: 
175:             $evalValue = true;
176:             while (null !== ($type = $tokenStream->getType())) {
177:                 $value = $tokenStream->getTokenValue();
178: 
179:                 if (!isset($acceptedTokens[$type])) {
180:                     break;
181:                 } elseif ($tokenStream->is(T_STRING) && !isset($acceptedStrings[strtolower($value)])) {
182:                     $evalValue = false;
183:                 }
184: 
185:                 $this->valueDefinition .= $value;

187:             }
188: 
189:             if (null === $type || (',' !== $value && ';' !== $value)) {
190:                 throw new Exception\Parse(sprintf('Invalid value definition: "%s".', $this->valueDefinition), Exception\Parse::PARSE_ELEMENT_ERROR);

192: 
193:             $this->valueDefinition = trim($this->valueDefinition);

195:             $this->valueAsLink = !$evalValue;
