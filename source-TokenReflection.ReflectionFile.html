 78:         if ($this->tokenStream->count() <= 1) {
 79:             // No PHP content
 80:             return;
 81:         }
 82: 
 83:         try {
 84:             if (!$this->tokenStream->is(T_OPEN_TAG)) {
 85:                 $this->namespaces[] = new ReflectionFileNamespace($this->tokenStream, $this->broker, $this);
 86:             } else {
 87:                 $this->tokenStream->skipWhitespaces();
 88: 
 89:                 while (null !== ($type = $this->tokenStream->getType())) {
 90:                     switch ($type) {
 91:                         case T_WHITESPACE:
 92:                         case T_DOC_COMMENT:
 93:                         case T_COMMENT:
 94:                             break;
 95:                         case T_DECLARE:
 96:                             $this->tokenStream
 97:                                 ->skipWhitespaces()
 98:                                 ->findMatchingBracket()
 99:                                 ->skipWhitespaces()
100:                                 ->skipWhitespaces(); // Intentionally twice
101:                             break;
102:                         case T_NAMESPACE:
103:                             break 2;
104:                         default:
105:                             $this->namespaces[] = new ReflectionFileNamespace($this->tokenStream, $this->broker, $this);
106:                             return $this;
107:                     }
108: 
109:                     $this->tokenStream->skipWhitespaces();
110:                 }
111: 
112:                 while (null !== ($type = $this->tokenStream->getType())) {
113:                     if (T_NAMESPACE === $type) {
114:                         $this->namespaces[] = new ReflectionFileNamespace($this->tokenStream, $this->broker, $this);
115:                     } else {
116:                         $this->tokenStream->skipWhitespaces();
117:                     }
118:                 }
119:             }
120:

122:             return $this;
123:         } catch (Exception $e) {
124:             throw new Exception\Parse('Could not parse file contents.', Exception\Parse::PARSE_CHILDREN_ERROR, $e);
